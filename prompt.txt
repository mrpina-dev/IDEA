let reasonsData = [];
let reasonsMap = null;
const REASONS_KEY = 'idea-entry-reasons';
const REASONS_TS_KEY = 'idea-entry-reasons-ts';
const REASONS_TTL = 5 * 60 * 1000;

function buildReasonsMap(list) {
  const map = { 'Award': [], 'Fine': [], 'Purchase': [], 'Payout': [] };
  list.forEach(r => {
    const active = r.active !== false && String(r.active).toLowerCase() !== 'no';
    if (!active) return;
    const t = getField(r, 'type', 'Type');
    const reason = getField(r, 'reason', 'Reason');
    if (map[t] && reason) map[t].push(reason);
  });
  map['Award - Held'] = map['Award'];
  return map;
}

function reasonsFor(type) {
  const t = (type === 'Award - Held') ? 'Award' : type;
  if (reasonsMap && reasonsMap[t] && reasonsMap[t].length) return reasonsMap[t];
  return REASONS[type] || REASONS[t] || [];
}

function fetchReasons() {
  try {
    const ts = localStorage.getItem(REASONS_TS_KEY);
    const raw = localStorage.getItem(REASONS_KEY);
    if (ts && raw && Date.now() - parseInt(ts,10) < REASONS_TTL) {
      reasonsData = JSON.parse(raw);
      reasonsMap = buildReasonsMap(reasonsData);
    }
  } catch(_) {}
  return fetch(API + '?action=getReasons&_=' + Date.now(), { cache: 'no-store' })
    .then(r => r.json())
    .then(data => {
      let list = Array.isArray(data) ? data : (data && Array.isArray(data.reasons) ? data.reasons : []);
      if (list.length) {
        reasonsData = list;
        reasonsMap = buildReasonsMap(list);
        try {
          localStorage.setItem(REASONS_KEY, JSON.stringify(list));
          localStorage.setItem(REASONS_TS_KEY, String(Date.now()));
        } catch(_) {}
      }
    })
    .catch(() => {});
}

let reasonsFilterType = 'All';
let showInactiveReasons = false;
let activeReasonEdit = null;

function reasonRow(r)    { return getField(r,'row','Row'); }
function reasonType(r)   { return getField(r,'type','Type') || ''; }
function reasonText(r)   { return getField(r,'reason','Reason') || ''; }
function reasonDetail(r) { return getField(r,'detail','Detail') || ''; }
function reasonActive(r) { const v = r.active !== undefined ? r.active : r.Active; return v !== false && String(v).toLowerCase() !== 'no'; }

async function refreshReasons() {
  try {
    const res = await fetch(API + '?action=getReasons&_=' + Date.now(), { cache: 'no-store' });
    const data = await res.json();
    let list = Array.isArray(data) ? data : (data && Array.isArray(data.reasons) ? data.reasons : []);
    reasonsData = list;
    reasonsMap = buildReasonsMap(list);
    try {
      localStorage.setItem(REASONS_KEY, JSON.stringify(list));
      localStorage.setItem(REASONS_TS_KEY, String(Date.now()));
    } catch(_) {}
  } catch(_) {}
}

document.getElementById('btn-reasons').addEventListener('click', async () => {
  showLoading('Loading reasons…');
  try { await refreshReasons(); }
  catch(e) { hideLoading(); toast('Failed to load reasons', 'error'); return; }
  hideLoading();
  reasonsFilterType = 'All';
  showInactiveReasons = false;
  buildReasonsFilter();
  renderReasonsList();
  showScreen('s-reasons');
});

document.getElementById('reasons-back').addEventListener('click', () => showScreen('s-home'));

document.getElementById('reasons-add-btn').addEventListener('click', () => openReasonEdit(null));

function buildReasonsFilter() {
  const row = document.getElementById('reasons-filter-row');
  const types = ['All','Award','Fine','Purchase','Payout'];
  row.innerHTML = types.map(t =>
    `<button class="wage-filter-btn${t === reasonsFilterType ? ' active' : ''}" data-type="${t}">${t}</button>`
  ).join('');
  row.querySelectorAll('.wage-filter-btn').forEach(btn => {
    btn.addEventListener('click', () => {
      reasonsFilterType = btn.dataset.type;
      buildReasonsFilter();
      renderReasonsList();
    });
  });
}

function renderReasonsList() {
  const types = reasonsFilterType === 'All' ? ['Award','Fine','Purchase','Payout'] : [reasonsFilterType];
  const active = reasonsData.filter(r => reasonActive(r));
  const inactive = reasonsData.filter(r => !reasonActive(r));
  let html = '';
  types.forEach(t => {
    const group = active.filter(r => reasonType(r) === t)
      .sort((a,b) => reasonText(a).localeCompare(reasonText(b)));
    if (!group.length) return;
    html += `<div class="section-group-head">${esc(t)}</div>`;
    html += group.map(reasonRowHTML).join('');
  });
  if (showInactiveReasons) {
    const inGroup = inactive.filter(r => reasonsFilterType === 'All' || reasonType(r) === reasonsFilterType)
      .sort((a,b) => reasonText(a).localeCompare(reasonText(b)));
    if (inGroup.length) {
      html += '<div class="section-group-head">Retired</div>';
      html += inGroup.map(reasonRowHTML).join('');
    }
  }
  if (!html) html = '<div class="manage-empty">No reasons found.</div>';
  document.getElementById('reasons-list').innerHTML = html;
  document.querySelectorAll('#reasons-list .manage-row[data-rrow]').forEach(el => {
    el.addEventListener('click', () => {
      const r = reasonsData.find(x => String(reasonRow(x)) === String(el.dataset.rrow));
      if (r) openReasonEdit(r);
    });
  });

  const wrap = document.getElementById('reasons-inactive-toggle-wrap');
  const togBtn = document.getElementById('reasons-inactive-toggle');
  if (inactive.length) {
    wrap.style.display = '';
    togBtn.textContent = showInactiveReasons ? `Hide Retired (${inactive.length})` : `Show Retired (${inactive.length})`;
  } else { wrap.style.display = 'none'; }
}

function reasonRowHTML(r) {
  const text = reasonText(r);
  const detail = reasonDetail(r);
  const active = reasonActive(r);
  return `<div class="manage-row" data-rrow="${esc(String(reasonRow(r)))}">
    <div class="manage-row-left">
      <div class="manage-row-name">${esc(text)}</div>
      ${detail ? `<div class="manage-row-sub">${esc(detail)}</div>` : ''}
    </div>
    <div class="manage-row-right">${!active ? '<span class="inactive-badge">Retired</span>' : ''}</div>
  </div>`;
}

document.getElementById('reasons-inactive-toggle').addEventListener('click', () => {
  showInactiveReasons = !showInactiveReasons;
  renderReasonsList();
});

function openReasonEdit(r) {
  activeReasonEdit = r;
  const isNew = !r;
  document.getElementById('reason-edit-sub').textContent = isNew ? 'New Reason' : 'Edit Reason';
  document.getElementById('reason-edit-type').value = isNew ? (reasonsFilterType !== 'All' ? reasonsFilterType : 'Award') : reasonType(r);
  document.getElementById('reason-edit-text').value = isNew ? '' : reasonText(r);
  document.getElementById('reason-edit-detail').value = isNew ? '' : reasonDetail(r);
  document.getElementById('reason-edit-deactivate').style.display = (!isNew && reasonActive(r)) ? '' : 'none';
  document.getElementById('reason-edit-reactivate').style.display = (!isNew && !reasonActive(r)) ? '' : 'none';
  showScreen('s-reason-edit');
}

document.getElementById('reason-edit-back').addEventListener('click', () => { renderReasonsList(); showScreen('s-reasons'); });

document.getElementById('reason-edit-save').addEventListener('click', async () => {
  const type = document.getElementById('reason-edit-type').value;
  const reason = document.getElementById('reason-edit-text').value.trim();
  const detail = document.getElementById('reason-edit-detail').value.trim();
  if (!reason) { toast('Reason text is required', 'error'); return; }
  showLoading('Saving…');
  try {
    let params;
    if (activeReasonEdit) params = { action:'updateReason', row: reasonRow(activeReasonEdit), type, reason, detail };
    else params = { action:'addReason', type, reason, detail };
    const res = await fetch(apiUrl(params));
    const data = await res.json();
    if (data && (data.success || data.status === 'ok' || data.result === 'success')) {
      toast('Reason saved ✓', 'success');
      await refreshReasons();
      renderReasonsList();
      showScreen('s-reasons');
    } else { toast('Error saving — try again', 'error'); }
  } catch(e) { toast('Network error', 'error'); }
  finally { hideLoading(); }
});

document.getElementById('reason-edit-deactivate').addEventListener('click', () => {
  if (!activeReasonEdit) return;
  confirm('Retire Reason', 'This reason will be hidden from new transactions everywhere. Existing transactions keep it.', async () => {
    showLoading('Retiring…');
    try {
      const res = await fetch(apiUrl({ action:'deactivateReason', row: reasonRow(activeReasonEdit) }));
      const data = await res.json();
      if (data && (data.success || data.status === 'ok' || data.result === 'success')) {
        toast('Reason retired', 'success');
        await refreshReasons();
        renderReasonsList();
        showScreen('s-reasons');
      } else { toast('Error — try again', 'error'); }
    } catch(e) { toast('Network error', 'error'); }
    finally { hideLoading(); }
  });
});

document.getElementById('reason-edit-reactivate').addEventListener('click', async () => {
  if (!activeReasonEdit) return;
  showLoading('Reactivating…');
  try {
    const res = await fetch(apiUrl({ action:'reactivateReason', row: reasonRow(activeReasonEdit) }));
    const data = await res.json();
    if (data && (data.success || data.status === 'ok' || data.result === 'success')) {
      toast('Reason reactivated ✓', 'success');
      await refreshReasons();
      renderReasonsList();
      showScreen('s-reasons');
    } else { toast('Error — try again', 'error'); }
  } catch(e) { toast('Network error', 'error'); }
  finally { hideLoading(); }
});
