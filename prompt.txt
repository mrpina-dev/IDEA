let payoutOwedList = [];

function renderPayout(data) {
  paidThisSession.clear();
  let owed = [];
  if (data && Array.isArray(data.owed)) {
    owed = data.owed.map(s => ({
      name: getField(s,'name','Name') || '—',
      sec:  getField(s,'section','Section') || '',
      bank: parseFloat(getField(s,'bankBalance','bank_balance','BankBalance') || 0)
    }));
  }
  owed = owed.filter(s => s.bank > 0).sort((a,b) => a.name.localeCompare(b.name));
  payoutOwedList = owed;

  const summaryEl = document.getElementById('payout-summary');
  const contentEl = document.getElementById('payout-content');

  if (!owed.length) {
    summaryEl.innerHTML = '<span style="color:var(--green)">All paid up - you do not owe any physical coins right now.</span>';
    contentEl.innerHTML = '';
    return;
  }

  const totalCoins = owed.reduce((sum, s) => sum + s.bank, 0);
  summaryEl.innerHTML =
    `<span style="color:var(--gold)">You owe physical coins to ${owed.length} student${owed.length!==1?'s':''}.</span><br>` +
    `<span style="color:var(--dim)">Total: ${totalCoins} i&#162; across all students.</span>`;

  contentEl.innerHTML =
    `<div class="payout-sec-head give">Owed Coins <span class="payout-sec-count">(${owed.length})</span></div>` +
    `<button class="payout-pay-all-btn" id="payout-pay-all-btn" onclick="handlePayAll()">Pay All (${owed.length}) Students</button>` +
    owed.map(s => renderPayoutOwedRow(s)).join('');
}

function renderPayoutOwedRow(s) {
  const isPaid = paidThisSession.has(s.name);
  return `<div class="payout-row${isPaid ? ' paid' : ''}" id="payout-row-${CSS.escape(s.name)}">` +
    `<div style="display:flex;flex-direction:column;flex:1;min-width:0;">` +
    `<span style="color:var(--white)">${esc(s.name)}</span>` +
    (s.sec ? `<span style="font-size:0.65rem;color:var(--dim);letter-spacing:0.04em;">${esc(s.sec)}</span>` : '') +
    `</div>` +
    `<span class="payout-bal pos" style="margin-right:0.65rem;">${Math.round(s.bank)} i&#162;</span>` +
    (isPaid
      ? `<span class="payout-paid-badge">PAID &#10003;</span>`
      : `<button class="payout-pay-btn" onclick="handlePayStudent(${JSON.stringify(s.name)})">PAY</button>`) +
    `</div>`;
}

function payStudentInternal(name) {
  const s = payoutOwedList.find(x => x.name === name);
  if (!s) return;
  const amount = Math.round(s.bank);
  if (amount <= 0) return;
  fetch(apiUrl({ action: 'logTransaction', name, type: 'Payout', reason: 'Coin Payout to Student', amount }))
    .catch(() => {});
  paidThisSession.add(name);
}

function handlePayStudent(name) {
  payStudentInternal(name);
  const row = document.getElementById('payout-row-' + CSS.escape(name));
  if (row) {
    row.classList.add('paid');
    const btn = row.querySelector('.payout-pay-btn');
    if (btn) { const badge = document.createElement('span'); badge.className = 'payout-paid-badge'; badge.textContent = 'PAID ✓'; btn.replaceWith(badge); }
  }
  checkAllPaid();
  toast('Payout logged for 1 student', 'success');
}

function handlePayAll() {
  const unpaid = payoutOwedList.filter(s => !paidThisSession.has(s.name));
  if (!unpaid.length) return;
  unpaid.forEach(s => {
    payStudentInternal(s.name);
    const row = document.getElementById('payout-row-' + CSS.escape(s.name));
    if (row) {
      row.classList.add('paid');
      const btn = row.querySelector('.payout-pay-btn');
      if (btn) { const badge = document.createElement('span'); badge.className = 'payout-paid-badge'; badge.textContent = 'PAID ✓'; btn.replaceWith(badge); }
    }
  });
  checkAllPaid();
  toast(`Payout logged for ${unpaid.length} student${unpaid.length !== 1 ? 's' : ''}`, 'success');
}

function checkAllPaid() {
  const allPaid = payoutOwedList.every(s => paidThisSession.has(s.name));
  const btn = document.getElementById('payout-pay-all-btn');
  if (btn && allPaid) { btn.textContent = 'All Paid ✓'; btn.disabled = true; }
}
