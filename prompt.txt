JAVASCRIPT - add a self-contained block modeled on the existing Reconcile Balance screen logic:

let acceptStudent = null;
const acceptSearchInput   = document.getElementById('accept-student-search');
const acceptSearchResults = document.getElementById('accept-search-results');
const acceptStudentCard   = document.getElementById('accept-student-card');

document.getElementById('btn-accept-coins').addEventListener('click', async () => {
  if (!await requireStudents()) return;
  resetAccept();
  showScreen('s-accept');
  acceptSearchInput.focus();
});

document.getElementById('accept-back').addEventListener('click', () => {
  resetAccept();
  showScreen('s-home');
});

acceptSearchInput.addEventListener('input', () => {
  const q = acceptSearchInput.value.trim().toLowerCase();
  if (!q) { acceptSearchResults.style.display = 'none'; return; }
  const matches = allStudents.filter(s => {
    const name = (getField(s,'name','Name','Full Name','Student') || '').toLowerCase();
    const sec  = (getField(s,'section','Section','Class') || '').toLowerCase();
    return name.includes(q) || sec.includes(q);
  }).slice(0, 10);
  if (!matches.length) {
    acceptSearchResults.innerHTML = '<div class="result-empty">No students found.</div>';
    acceptSearchResults.style.display = 'block';
    return;
  }
  acceptSearchResults.innerHTML = matches.map(s => {
    const name = getField(s,'name','Name','Full Name','Student') || '—';
    const sec  = getField(s,'section','Section','Class') || '—';
    const bal  = studentBal(s);
    return `<div class="result-item" tabindex="0" data-name="${esc(name)}" data-sec="${esc(sec)}" data-bal="${bal}">
      <div class="result-info"><span class="result-name">${esc(name)}</span><span class="result-section">${esc(sec)}</span></div>
      <span class="result-bal ${bal>=0?'pos':'neg'}">${Math.round(bal)} i&#162;</span></div>`;
  }).join('');
  acceptSearchResults.style.display = 'block';
  acceptSearchResults.querySelectorAll('.result-item').forEach(el => {
    el.addEventListener('click', () => acceptPickStudent(el.dataset.name, el.dataset.sec, parseFloat(el.dataset.bal)));
  });
});

function acceptPickStudent(name, section, balance) {
  acceptStudent = { name, section, balance };
  acceptSearchInput.value = '';
  acceptSearchResults.style.display = 'none';
  document.getElementById('accept-search-wrap').style.display = 'none';
  document.getElementById('accept-card-name').textContent = name;
  document.getElementById('accept-card-section').textContent = section;
  const elig = document.getElementById('accept-card-elig');
  elig.textContent = balance < 0 ? 'IN DEBT' : 'ELIGIBLE';
  elig.className = 'elig-badge ' + (balance < 0 ? 'bad' : 'ok');
  const balEl = document.getElementById('accept-card-bal');
  balEl.textContent = Math.round(balance) + ' i¢';
  balEl.className = 'student-card-bal ' + (balance >= 0 ? 'pos' : 'neg');
  acceptStudentCard.classList.add('show');
  document.getElementById('accept-step2').style.display = '';
  document.getElementById('accept-coins-input').value = '';
  document.getElementById('accept-preview').textContent = '';
  document.getElementById('accept-submit-btn').disabled = true;
}

document.getElementById('accept-change-student').addEventListener('click', () => {
  acceptStudent = null;
  acceptStudentCard.classList.remove('show');
  document.getElementById('accept-search-wrap').style.display = '';
  acceptSearchInput.value = '';
  acceptSearchInput.focus();
  document.getElementById('accept-step2').style.display = 'none';
});

document.getElementById('accept-coins-input').addEventListener('input', () => {
  if (!acceptStudent) return;
  const v = parseInt(document.getElementById('accept-coins-input').value, 10);
  const preview = document.getElementById('accept-preview');
  const btn = document.getElementById('accept-submit-btn');
  if (isNaN(v) || v <= 0) { preview.textContent = ''; btn.disabled = true; return; }
  const newBal = Math.round(acceptStudent.balance) + v;
  preview.textContent = `New balance: ${newBal} i¢`;
  btn.disabled = false;
});

document.getElementById('accept-submit-btn').addEventListener('click', () => {
  if (!acceptStudent) return;
  const v = parseInt(document.getElementById('accept-coins-input').value, 10);
  if (isNaN(v) || v <= 0) return;
  const name = acceptStudent.name;
  const raw = allStudents.find(s => (getField(s,'name','Name','Full Name','Student') || '').toLowerCase() === name.toLowerCase());
  if (raw) {
    const k = raw.awarded !== undefined ? 'awarded' : 'Awarded';
    raw[k] = (parseFloat(raw[k]) || 0) + v;
  }
  toast(`Accepted ${v} i¢ from ${name}`, 'success');
  resetAccept();
  showScreen('s-home');
  fetch(apiUrl({ action: 'logTransaction', name, type: 'Award', reason: 'Physical Coin Submission', amount: v }))
    .then(r => r.json())
    .then(d => { if (!d || !(d.success || d.status === 'ok' || d.result === 'success')) toast('Sync failed — check sheet', 'error'); })
    .catch(() => toast('Sync failed — check sheet', 'error'));
});

function resetAccept() {
  acceptStudent = null;
  acceptSearchInput.value = '';
  acceptSearchResults.style.display = 'none';
  document.getElementById('accept-search-wrap').style.display = '';
  acceptStudentCard.classList.remove('show');
  document.getElementById('accept-step2').style.display = 'none';
  document.getElementById('accept-coins-input').value = '';
  document.getElementById('accept-preview').textContent = '';
  document.getElementById('accept-body').scrollTop = 0;
}
